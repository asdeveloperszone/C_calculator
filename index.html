void leapyear()
{
	int userInputYear;
	printf("Enter year:");
	scanf("%d", &userInputYear);
	if (userInputYear % 4 == 0)
	{
		printf("%d is a leap Year.", userInputYear);
	}
	else
	{
		printf("%d is not a leap Year.", userInputYear);
	}
}
void calculator()
{
	float stNumber, ndNumber;
	char userChoice;
	printf("Enter 1st Number:\t");
	scanf("%f", &stNumber);
	printf("Enter 2nd Number:\t");
	scanf("%f", &ndNumber);
	printf("Enter choice:(+,-,*,/)");
	scanf(" %c", &userChoice);
	switch (userChoice)
	{	
case '+':
	printf("%.2f + %.2f = %.2f", stNumber, ndNumber, stNumber + ndNumber);
case '-':
	printf("%.2f - %.2f = %.2f", stNumber, ndNumber, stNumber - ndNumber);
break ;
case '*':

	printf("%.2f * %.2f = %.2f", stNumber, ndNumber, stNumber * ndNumber);
break ;
case '/':
	if (stNumber != 0 && ndNumber != 0)
	{
		printf("%.2f รท %.2f = %.2f", stNumber, ndNumber, stNumber / ndNumber);
		break;
	}
	else
	{
		printf("Division by 0 is Invalid.");
	}}
}
